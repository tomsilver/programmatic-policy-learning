from prpl_utils.search import run_astar, SearchMetrics


class GeneratedPolicy:
    """
    Purely reactive wall-follower with a small amount of "void navigation" logic
    to reliably reach the entrance (-1, 0) from arbitrary outer-void starts.

    Inside the maze it uses a left-hand rule (with a bit of goal bias). This is
    cheap but can be suboptimal and can fail in mazes that are not simply connected.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.goal = None
        self.mode = "to_entrance"  # then "wall_follow"
        self.last_action = 1  # pretend we were moving South initially

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.mode = "to_entrance"
        self.last_action = 1

    @staticmethod
    def _manhattan(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _valid(self, s, a):
        ns = self.get_next_state(s, a)
        return ns != s

    def _step_toward(self, s, target):
        # Greedy move that reduces manhattan distance, but only if valid.
        # Tie-break with a fixed preference order to keep it deterministic.
        r, c = s
        tr, tc = target
        prefs = []
        if tr < r:
            prefs.append(0)
        elif tr > r:
            prefs.append(1)
        if tc > c:
            prefs.append(2)
        elif tc < c:
            prefs.append(3)

        # Fill remaining actions deterministically
        for a in [0, 1, 2, 3]:
            if a not in prefs:
                prefs.append(a)

        for a in prefs:
            if self._valid(s, a):
                return a
        return 0  # fallback; should never happen in free space but always valid integer

    def _to_entrance_action(self, obs):
        # The entrance is (-1, 0) and must be approached from (-2, 0).
        # Key hazard: row -1 is wall except col 0; being at (-1, x!=0) is a trap.
        r, c = obs

        # If already at entrance, go South into maze
        if obs == (-1, 0):
            return 1

        # If accidentally at row -1 but not at col 0, escape by going North.
        # (East/West blocked; South usually blocked by wall border.)
        if r == -1 and c != 0:
            return 0

        # Strategy in the void: get to row <= -2, align to col 0, then go to (-2,0), then South.
        if r == -2 and c == 0:
            return 1  # step into entrance at (-1,0)

        # Prefer to ensure we're in safe horizontal-movement band (row <= -2) before aligning columns.
        if r > -2:
            # Move North until at row -2 or above
            return 0

        # Now r <= -2: safe to move east/west without hitting row -1 wall.
        if c != 0:
            return 3 if c > 0 else 2

        # c == 0 and r <= -3: move South toward -2
        if r < -2:
            return 1

        # Fallback: greedy to (-2,0)
        return self._step_toward(obs, (-2, 0))

    def _wall_follow_action(self, obs):
        # Left-hand rule with a mild goal bias:
        # try left, then forward, then right, then back relative to last_action.
        # If multiple are possible, choose the one that most reduces distance to goal.
        if self.goal is None:
            goal = (0, 0)
        else:
            goal = self.goal

        # relative directions based on last_action
        left_of = {0: 3, 1: 2, 2: 0, 3: 1}
        right_of = {0: 2, 1: 3, 2: 1, 3: 0}
        back_of = {0: 1, 1: 0, 2: 3, 3: 2}

        candidates = [left_of[self.last_action], self.last_action, right_of[self.last_action], back_of[self.last_action]]
        valid = [a for a in candidates if self._valid(obs, a)]
        if not valid:
            return 0

        # Choose among valid using goal distance tie-breaker
        best_a = valid[0]
        best_d = self._manhattan(self.get_next_state(obs, best_a), goal)
        for a in valid[1:]:
            d = self._manhattan(self.get_next_state(obs, a), goal)
            if d < best_d:
                best_d = d
                best_a = a
        return best_a

    def get_action(self, obs):
        # Switch mode once we enter the maze (row >= 0 and col >= 0 typically).
        if self.mode == "to_entrance":
            a = self._to_entrance_action(obs)
            self.last_action = a
            # If we just moved into the maze or are at/inside, switch.
            ns = self.get_next_state(obs, a)
            if ns[0] >= 0 and ns[1] >= 0:
                self.mode = "wall_follow"
            return a

        a = self._wall_follow_action(obs)
        self.last_action = a
        return a

class GeneratedPolicy:
    """
    Planning-heavy: always plans from the current observation with A* (no caching),
    using a Manhattan heuristic to the provided goal.

    Very robust but computationally expensive.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None

    def reset(self, obs, info):
        self.goal = info.get("goal", None)

    def _get_successors(self, state):
        for action in self.get_actions():
            ns = self.get_next_state(state, action)
            if ns != state:
                yield (action, ns, float(self.get_cost()))

    def _heuristic(self, state):
        if self.goal is None:
            return 0.0
        return float(abs(state[0] - self.goal[0]) + abs(state[1] - self.goal[1]))

    def get_action(self, obs):
        # Plan every step from scratch.
        if self.goal is None:
            # If goal is missing, just move South (still valid integer).
            return 1

        def is_goal(s):
            return self.check_goal_env(s, self.goal)

        states, actions, metrics = run_astar(
            initial_state=obs,
            check_goal=is_goal,
            get_successors=self._get_successors,
            heuristic=self._heuristic,
            max_expansions=2_000_000,
            max_evals=2_000_000,
            timeout=2.0,
            lazy_expansion=False,
        )

        if actions:
            return actions[0]

        # Fallback: choose any valid action; if none change state, return 0.
        for a in self.get_actions():
            if self.get_next_state(obs, a) != obs:
                return a
        return 0

class GeneratedPolicy:
    """
    Hybrid: mostly greedy toward goal, but triggers A* when greedy gets stuck
    (no improving move) or when it cycles. Uses short-horizon A* to escape local traps.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None
        self.plan = []
        self.expected_obs = None

        self.recent = []
        self.recent_max = 30

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None
        self.recent = [obs]

    def _manhattan(self, s):
        if self.goal is None:
            return 0
        return abs(s[0] - self.goal[0]) + abs(s[1] - self.goal[1])

    def _valid_actions(self, obs):
        acts = []
        for a in self.get_actions():
            if self.get_next_state(obs, a) != obs:
                acts.append(a)
        return acts

    def _greedy_action(self, obs):
        # Pick valid action that most reduces manhattan distance; if none reduce, return best non-increasing.
        valid = self._valid_actions(obs)
        if not valid:
            return 0
        d0 = self._manhattan(obs)
        best = valid[0]
        best_d = self._manhattan(self.get_next_state(obs, best))
        for a in valid[1:]:
            d = self._manhattan(self.get_next_state(obs, a))
            if d < best_d:
                best, best_d = a, d
        return best

    def _get_successors(self, state):
        for action in self.get_actions():
            ns = self.get_next_state(state, action)
            if ns != state:
                yield (action, ns, float(self.get_cost()))

    def _heuristic(self, state):
        return float(self._manhattan(state))

    def _need_plan(self, obs):
        # Trigger planning if:
        # 1) Greedy does not strictly improve distance, OR
        # 2) We have visited this obs recently (cycle indicator).
        if self.goal is None:
            return False
        a = self._greedy_action(obs)
        ns = self.get_next_state(obs, a)
        if self._manhattan(ns) >= self._manhattan(obs):
            return True
        if obs in self.recent[-10:]:
            return True
        return False

    def _plan_from(self, obs):
        if self.goal is None:
            self.plan = []
            self.expected_obs = obs
            return

        def is_goal(s):
            return self.check_goal_env(s, self.goal)

        # Use a moderate budget; this is invoked only when needed.
        states, actions, metrics = run_astar(
            initial_state=obs,
            check_goal=is_goal,
            get_successors=self._get_successors,
            heuristic=self._heuristic,
            max_expansions=500_000,
            max_evals=500_000,
            timeout=0.8,
            lazy_expansion=True,
        )
        self.plan = list(actions) if actions else []
        self.expected_obs = obs

    def get_action(self, obs):
        # Update recent history
        self.recent.append(obs)
        if len(self.recent) > self.recent_max:
            self.recent.pop(0)

        # Follow cached plan if consistent
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Decide whether to plan
        if self._need_plan(obs):
            self._plan_from(obs)
            if self.plan and self.expected_obs == obs:
                a = self.plan.pop(0)
                self.expected_obs = self.get_next_state(obs, a)
                return a

        # Otherwise act greedily
        a = self._greedy_action(obs)
        return a

class GeneratedPolicy:
    """
    Hybrid with explicit two-stage planning:

    1) Reactive navigation in the outer void to reach the entrance (-1,0).
    2) Inside the maze, repeated *bounded* A* "receding horizon": plan, execute
       a small prefix, then replan. This is robust to small model mismatches and
       keeps memory bounded, at the cost of more frequent planning than full caching.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None
        self.phase = "to_entrance"

        self.plan = []
        self.expected_obs = None
        self.prefix_len = 6  # execute at most this many steps before replanning

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.phase = "to_entrance"
        self.plan = []
        self.expected_obs = None

    def _to_entrance_action(self, obs):
        r, c = obs
        if obs == (-1, 0):
            return 1
        if r == -1 and c != 0:
            return 0
        if r > -2:
            return 0
        if c != 0:
            return 3 if c > 0 else 2
        if r < -2:
            return 1
        return 1

    def _get_successors(self, state):
        for action in self.get_actions():
            ns = self.get_next_state(state, action)
            if ns != state:
                yield (action, ns, float(self.get_cost()))

    def _heuristic(self, state):
        if self.goal is None:
            return 0.0
        return float(abs(state[0] - self.goal[0]) + abs(state[1] - self.goal[1]))

    def _plan_prefix(self, obs):
        if self.goal is None:
            self.plan = []
            self.expected_obs = obs
            return

        def is_goal(s):
            return self.check_goal_env(s, self.goal)

        states, actions, metrics = run_astar(
            initial_state=obs,
            check_goal=is_goal,
            get_successors=self._get_successors,
            heuristic=self._heuristic,
            max_expansions=800_000,
            max_evals=800_000,
            timeout=1.2,
            lazy_expansion=True,
        )
        if actions:
            self.plan = list(actions[: self.prefix_len])
        else:
            self.plan = []
        self.expected_obs = obs

    def get_action(self, obs):
        if self.phase == "to_entrance":
            a = self._to_entrance_action(obs)
            ns = self.get_next_state(obs, a)
            if ns[0] >= 0 and ns[1] >= 0:
                self.phase = "receding_astar"
                self.plan = []
                self.expected_obs = None
            return a

        # Execute current prefix if consistent
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Otherwise, (re)plan a short prefix
        self._plan_prefix(obs)
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Fallback: any valid move
        for a in self.get_actions():
            if self.get_next_state(obs, a) != obs:
                return a
        return 0

class GeneratedPolicy:
    """
    Planning-heavy with "goal-conditioned waypoints":

    - Stage 1: plan (with A*) to the entrance (-1,0) from wherever we are in the void.
      This is robust even if the outer void is large because the heuristic guides directly.
    - Stage 2: plan from (0,0) (or current state) to the true goal.

    Uses caching for each stage, and replans if the plan is invalidated.
    This is more planning-heavy than reactive-entrance variants, but can recover
    from unusual positions near the border.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None
        self.stage = 1  # 1: to entrance, 2: to goal
        self.plan = []
        self.expected_obs = None

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.stage = 1
        self.plan = []
        self.expected_obs = None

    def _get_successors(self, state):
        for action in self.get_actions():
            ns = self.get_next_state(state, action)
            if ns != state:
                yield (action, ns, float(self.get_cost()))

    def _heuristic_to(self, target):
        def h(state):
            return float(abs(state[0] - target[0]) + abs(state[1] - target[1]))
        return h

    def _plan_to(self, obs, target, timeout, max_exp):
        states, actions, metrics = run_astar(
            initial_state=obs,
            check_goal=lambda s: s == target,
            get_successors=self._get_successors,
            heuristic=self._heuristic_to(target),
            max_expansions=max_exp,
            max_evals=max_exp,
            timeout=timeout,
            lazy_expansion=False,
        )
        self.plan = list(actions) if actions else []
        self.expected_obs = obs

    def _plan_to_goal(self, obs):
        if self.goal is None:
            self.plan = []
            self.expected_obs = obs
            return
        states, actions, metrics = run_astar(
            initial_state=obs,
            check_goal=lambda s: self.check_goal_env(s, self.goal),
            get_successors=self._get_successors,
            heuristic=self._heuristic_to(self.goal),
            max_expansions=5_000_000,
            max_evals=5_000_000,
            timeout=4.0,
            lazy_expansion=False,
        )
        self.plan = list(actions) if actions else []
        self.expected_obs = obs

    def get_action(self, obs):
        # Stage transition detection
        if self.stage == 1:
            # Once at or inside maze, move to stage 2.
            if obs[0] >= 0 and obs[1] >= 0:
                self.stage = 2
                self.plan = []
                self.expected_obs = None

        # Follow cached plan if consistent
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            # Update stage if we crossed into the maze
            if self.stage == 1 and self.expected_obs[0] >= 0 and self.expected_obs[1] >= 0:
                self.stage = 2
                self.plan = []
                self.expected_obs = None
            return a

        # Need to (re)plan
        if self.stage == 1:
            # Plan to the entrance (-1,0), then next step typically South to (0,0).
            # Use a moderate budget; heuristic strongly guides through free void.
            self._plan_to(obs, (-1, 0), timeout=1.5, max_exp=1_000_000)
            if not self.plan:
                # If planning fails, use a safe reactive fallback to avoid row -1 traps.
                r, c = obs
                if r == -1 and c != 0:
                    return 0
                if r > -2:
                    return 0
                if c != 0:
                    return 3 if c > 0 else 2
                if r < -2:
                    return 1
                return 1

        else:
            self._plan_to_goal(obs)

        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Final fallback
        for a in self.get_actions():
            if self.get_next_state(obs, a) != obs:
                return a
        return 0
