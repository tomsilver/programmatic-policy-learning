class GeneratedPolicy:
    """
    Purely reactive: two-phase navigation.
    1) In the outer void, go to the entrance approach cell (-2, 0) then step into (-1,0) and (0,0).
       Must avoid row -1 except at col 0 (otherwise stuck).
    2) Inside the maze, greedily reduce Manhattan distance to goal, with simple blocked-move fallback.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.goal = None
        self.last_obs = None
        self.last_action = 0
        self.stuck_count = 0
        self.phase = "void_to_entrance"

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.last_obs = obs
        self.last_action = 0
        self.stuck_count = 0
        self.phase = "void_to_entrance" if obs[0] < 0 or obs[1] < 0 else "inside"

    def _move_towards(self, obs, target):
        r, c = obs
        tr, tc = target

        # Prefer reducing row distance, then col distance (arbitrary tie-break).
        candidates = []
        if tr < r:
            candidates.append(0)  # N
        elif tr > r:
            candidates.append(1)  # S
        if tc > c:
            candidates.append(2)  # E
        elif tc < c:
            candidates.append(3)  # W

        # Add remaining actions as fallback to keep it total.
        for a in self.get_actions():
            if a not in candidates:
                candidates.append(a)
        return candidates

    def _choose_first_valid(self, obs, actions_pref):
        for a in actions_pref:
            ns = self.get_next_state(obs, a)
            if ns != obs:
                return a
        # If fully blocked (should be rare), return something valid.
        return 0

    def get_action(self, obs):
        # Track stuckness.
        if self.last_obs == obs:
            self.stuck_count += 1
        else:
            self.stuck_count = 0
        self.last_obs = obs

        r, c = obs

        # Phase selection
        if r >= 0 and c >= 0:
            self.phase = "inside"

        if self.phase == "void_to_entrance":
            # Critical safety: never remain on row -1 at nonzero col; if we are, attempt to go North/South off it.
            if r == -1 and c != 0:
                # East/West won't work; try moving away from the wall row.
                for a in (0, 1):  # N, S
                    ns = self.get_next_state(obs, a)
                    if ns != obs:
                        self.last_action = a
                        return a
                self.last_action = 0
                return 0

            # Navigate to (-2,0), then to (-1,0), then into maze.
            if obs == (-2, 0):
                # Step to entrance
                a = 1  # South to (-1,0)
                if self.get_next_state(obs, a) != obs:
                    self.last_action = a
                    return a
                # fallback
                self.last_action = 0
                return 0

            if obs == (-1, 0):
                a = 1  # South to (0,0)
                if self.get_next_state(obs, a) != obs:
                    self.last_action = a
                    return a
                self.last_action = 0
                return 0

            # From anywhere in void, approach (-2,0).
            pref = self._move_towards(obs, (-2, 0))
            a = self._choose_first_valid(obs, pref)
            self.last_action = a
            return a

        # Inside maze: greedy to goal with simple fallback.
        goal = self.goal
        if goal is None:
            # Shouldn't happen, but keep total.
            return 0

        pref = self._move_towards(obs, goal)
        # If we've been stuck repeating, rotate preference to escape local traps a bit.
        if self.stuck_count >= 2:
            # Cycle the action order deterministically.
            pref = pref[1:] + pref[:1]
        a = self._choose_first_valid(obs, pref)
        self.last_action = a
        return a

            return a

    - In outer void: reactive path to entrance approach (-2,0) avoiding row -1 traps.
    - Once inside (row>=0,col>=0): compute an A* plan to goal and cache it.
      Replan if deviated or if an action fails to move as expected.

        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.plan = []
        self.expected_obs = None
        self.inside = False

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None
        self.inside = (obs[0] >= 0 and obs[1] >= 0)

    def _heuristic(self, state):
        if self.goal is None:
            return 0.0
        return float(abs(state[0] - self.goal[0]) + abs(state[1] - self.goal[1]))

    def _get_successors(self, state):
        for action in self.get_actions():
            ns = self.get_next_state(state, action)
            if ns != state:
                yield (action, ns, float(self.get_cost()))

    def _plan_inside(self, obs):

            self.plan = []
            self.expected_obs = obs
            return
        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=lambda s: self.check_goal(s, self.goal),
            get_successors=self._get_successors,
            heuristic=self._heuristic,
            max_expansions=3_000_000,
            max_evals=3_000_000,
            timeout=2.0,
            lazy_expansion=True,

        self.expected_obs = obs

        # If on row -1 at wrong col, only N/S can help.

            for a in (0, 1):
                if self.get_next_state(obs, a) != obs:
                    return a
            return 0

        # If at (-1,0), go South into maze.
        if obs == (-1, 0):
            return 1 if self.get_next_state(obs, 1) != obs else 0

        # Aim for (-2,0), then step South.
        target = (-2, 0)

        # Greedy toward target with validity checks.
        prefs = []
        if target[0] < r:
            prefs.append(0)
        elif target[0] > r:
            prefs.append(1)
        if target[1] > c:
            prefs.append(2)
        elif target[1] < c:
            prefs.append(3)
        for a in self.get_actions():
            if a not in prefs:
                prefs.append(a)

        for a in prefs:
            if self.get_next_state(obs, a) != obs:
                return a
        return 0

        if obs[0] >= 0 and obs[1] >= 0:
            self.inside = True

            self.plan = []
            self.expected_obs = None
            return self._react_to_entrance(obs)

        # Inside: follow cached plan if consistent.

            a = self.plan.pop(0)
            ns = self.get_next_state(obs, a)
            # If action did not move, invalidate and replan next call.

                self.plan = []
                self.expected_obs = None
                # Immediate fallback
                for b in self.get_actions():
                    if self.get_next_state(obs, b) != obs:
                        return b
                return 0
            self.expected_obs = ns
            return a

        # Need a plan from current state.
        self._plan_inside(obs)
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Planning failed: greedy step to goal as fallback.
        if self.goal is not None:
            gr, gc = self.goal
            r, c = obs
            greedy = []
            if gr < r:
                greedy.append(0)
            elif gr > r:
                greedy.append(1)
            if gc > c:
                greedy.append(2)
            elif gc < c:
                greedy.append(3)
            for a in self.get_actions():
                if a not in greedy:
                    greedy.append(a)
            for a in greedy:
                if self.get_next_state(obs, a) != obs:
                    return a
        return 0

    Hybrid "reactive unless stuck":
    - Uses greedy Manhattan descent both outside (to entrance) and inside (to goal).
    - Detects local failure (no progress / repeated states) and then runs A* to recover.
    - Caches the A* plan and follows it strictly until consumed or invalidated.

        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.plan = []
        self.expected_obs = None

        self.prev_obs = None
        self.no_progress = 0

        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None
        self.prev_obs = obs
        self.no_progress = 0

    def _get_successors(self, state):

            if ns != state:
                yield (a, ns, float(self.get_cost()))

            return 0

        tr, tc = target

        # Special handling to avoid row -1 trap.

            for a in (0, 1):
                if self.get_next_state(obs, a) != obs:
                    return a
            return 0
        if obs == (-1, 0):
            return 1 if self.get_next_state(obs, 1) != obs else 0

        prefs = []
        # Choose move that most reduces manhattan distance (try all, sort by resulting h).
        scored = []
        for a in self.get_actions():
            ns = self.get_next_state(obs, a)
            if ns != obs:
                scored.append((self._heuristic(ns, target), a))
        scored.sort(key=lambda x: x[0])
        prefs = [a for _, a in scored]

        if prefs:
            return prefs[0]

        # If all blocked, return a valid action anyway.
        return 0

        if target is None:
            self.plan = []
            self.expected_obs = obs
            return

            initial_state=obs,

        self.expected_obs = obs

            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Progress tracking (simple).
        if self.prev_obs == obs:
            self.no_progress += 1
        else:
            self.no_progress = 0
        self.prev_obs = obs

        target = self._reactive_target(obs)

        # If we've failed to move for a bit, trigger planning.

            self._plan(obs, target)
            if self.plan and self.expected_obs == obs:
                a = self.plan.pop(0)
                self.expected_obs = self.get_next_state(obs, a)
                return a

        # Otherwise act reactively.
        a = self._reactive_action(obs, target)

        # If reactive action doesn't move, escalate immediately to planning next.
        if self.get_next_state(obs, a) == obs:
            self.no_progress = max(self.no_progress, 2)

        return a

    Planning-heavy but efficient: incremental A* replanning with a plan prefix reuse attempt.
    - Maintains a cached plan and expected state.
    - If deviated, replans; if still on-track, continues.

        self.get_next_state = get_next_state

        self.plan = []
        self.expected_obs = None

        self.plan = []
        self.expected_obs = None

            return 0.0
        man = abs(state[0] - self.goal[0]) + abs(state[1] - self.goal[1])
        return float(self.w * man)

    def _get_successors(self, state):
        c = float(self.get_cost())
        for a in self.get_actions():
            ns = self.get_next_state(state, a)
            if ns != state:
                yield (a, ns, c)

    def _replan(self, obs):
        if self.goal is None:
            self.plan = []
            self.expected_obs = obs
            return

        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=lambda s: self.check_goal(s, self.goal),
            get_successors=self._get_successors,
            heuristic=self._heuristic,

            max_evals=1_200_000,

        self.plan = list(actions) if actions is not None else []
        self.expected_obs = obs

    def get_action(self, obs):
        # Continue plan if valid.
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            ns = self.get_next_state(obs, a)
            # If failed, invalidate and replan.
            if ns == obs:
                self.plan = []
                self.expected_obs = None
            else:
                self.expected_obs = ns
                return a

        # Replan.
        self._replan(obs)

        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Fallback: choose any valid move (prefer S/E to eventually get into maze / toward goal).
        for a in (1, 2, 0, 3):
            if self.get_next_state(obs, a) != obs:
                return a
        return 0

class GeneratedPolicy:
    """
    Exploration-informed hybrid:
    - Maintains a lightweight memory of blocked transitions observed from visited states.
    - Uses reactive greedy control most of the time.
    - If greedy is predicted blocked (from memory) or repeatedly fails, runs A* with successors
      filtered by known-blocked actions to speed search and reduce thrashing.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.goal = None
        self.plan = []
        self.expected_obs = None

        # memory: state -> set(actions) that are known to be blocked from that state
        self.blocked = {}

        self.last_obs = None
        self.fail_streak = 0

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None
        self.blocked = {}
        self.last_obs = obs
        self.fail_streak = 0

    def _remember_transition(self, s, a, ns):
        if ns == s:
            self.blocked.setdefault(s, set()).add(a)

    def _heuristic(self, s, tgt):
        return float(abs(s[0] - tgt[0]) + abs(s[1] - tgt[1]))

    def _target(self, obs):
        if obs[0] >= 0 and obs[1] >= 0:
            return self.goal
        return (-2, 0)

    def _get_successors_filtered(self, state):
        blocked_actions = self.blocked.get(state, set())
        step_cost = float(self.get_cost())
        for a in self.get_actions():
            if a in blocked_actions:
                continue
            ns = self.get_next_state(state, a)
            if ns != state:
                yield (a, ns, step_cost)
            else:
                # Learn immediately during search calls too.
                self.blocked.setdefault(state, set()).add(a)

    def _plan(self, obs, target):
        if target is None:
            self.plan = []
            self.expected_obs = obs
            return
        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=(lambda s: self.check_goal(s, target)) if target == self.goal else (lambda s: s == target),
            get_successors=self._get_successors_filtered,
            heuristic=lambda s: self._heuristic(s, target),
            max_expansions=1_800_000,
            max_evals=1_800_000,
            timeout=1.8,
            lazy_expansion=True,
        )
        self.plan = list(actions) if actions is not None else []
        self.expected_obs = obs

    def _greedy_ranked_actions(self, obs, target):
        # Special entrance safety
        if obs[0] == -1 and obs[1] != 0:
            return [0, 1, 2, 3]
        if obs == (-1, 0):
            return [1, 0, 2, 3]

        scored = []
        for a in self.get_actions():
            if a in self.blocked.get(obs, set()):
                continue
            ns = self.get_next_state(obs, a)
            self._remember_transition(obs, a, ns)
            if ns != obs:
                scored.append((self._heuristic(ns, target), a))
        scored.sort(key=lambda x: x[0])
        ranked = [a for _, a in scored]

        # Append any remaining actions to keep it total.
        for a in self.get_actions():
            if a not in ranked:
                ranked.append(a)
        return ranked

    def get_action(self, obs):
        # Follow cached plan if valid.
