from prpl_utils.search import run_astar, SearchMetrics


class GeneratedPolicy:
    """
    Purely reactive policy:
    - In the outer void: go to the entrance approach point (-2, 0) safely, then to (-1, 0), then into (0,0).
    - Inside the maze: greedy Manhattan descent toward the goal with a simple blocked-move fallback.
    No global planning. Fast but can fail in complex mazes.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.goal = None
        self.last_obs = None
        self.last_action = 0
        self.blocked_streak = 0

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.last_obs = obs
        self.last_action = 0
        self.blocked_streak = 0

    def _manhattan(self, a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _is_blocked(self, obs, action):
        return self.get_next_state(obs, action) == obs

    def _outer_void_action(self, obs):
        r, c = obs

        # Always approach entrance from (-2, 0), then (-1, 0).
        # In the void, movement is obstacle-free, but row -1 is a wall except at col 0
        # and is "sticky" (no E/W). So avoid stepping onto row -1 unless at col 0.
        if (r, c) == (-2, 0):
            return 0  # North to (-3,0)? Actually to reach entrance, go South to (-1,0) from (-2,0)
        if (r, c) == (-2, 0):
            return 1

        # First, get to col 0 while staying on rows <= -2 to avoid wall row -1.
        if r == -1 and c != 0:
            # Stuck situation; try to go North off the wall row if possible.
            if not self._is_blocked(obs, 0):
                return 0
            return 1 if not self._is_blocked(obs, 1) else 2

        if r > -2:
            # Too close to wall row -1; back off to row -2
            return 0

        if c < 0:
            return 2
        if c > 0:
            return 3

        # Now at (r<=-2, c==0): move South until reaching (-1,0), then South to (0,0)
        if r < -2:
            return 1  # South
        if r == -2:
            return 1  # to (-1,0)
        return 1

    def _inner_greedy_action(self, obs):
        # Greedy toward goal, with tie-breaking and simple blocked fallback.
        if self.goal is None:
            # Fallback: always move East then South preference
            candidates = [2, 1, 0, 3]
        else:
            gr, gc = self.goal
            r, c = obs
            # Prefer reducing Manhattan distance. Choose among valid moves.
            actions = self.get_actions()
            scored = []
            for a in actions:
                ns = self.get_next_state(obs, a)
                if ns == obs:
                    continue
                d = self._manhattan(ns, (gr, gc))
                scored.append((d, a))
            if scored:
                scored.sort()
                return scored[0][1]
            candidates = [2, 1, 0, 3]

        # If all blocked (unlikely), return something valid.
        for a in candidates:
            if not self._is_blocked(obs, a):
                return a
        return 0

    def get_action(self, obs):
        # Track blockage streak
        if self.last_obs is not None:
            if self.get_next_state(self.last_obs, self.last_action) == self.last_obs:
                self.blocked_streak += 1
            else:
                self.blocked_streak = 0

        # Determine if we are outside (void) or in/at entrance region:
        # In the maze, rows >= 0 and cols >= 0. Entrance is (-1,0).
        r, c = obs
        if r < 0 or c < 0:
            # Outer void / border area
            # Special-case: at (-2,0), go South
            if obs == (-2, 0):
                action = 1
            else:
                action = self._outer_void_action(obs)
        else:
            action = self._inner_greedy_action(obs)

        # Ensure valid action integer 0..3 (and try to avoid blocked if possible)
        if action not in [0, 1, 2, 3]:
            action = 0
        if self.get_next_state(obs, action) == obs:
            for a in [0, 1, 2, 3]:
                if self.get_next_state(obs, a) != obs:
                    action = a
                    break

        self.last_obs = obs
        self.last_action = action
        return action

            return 0
        if c < 0:

        if c > 0:
            return 3

            self.expected_obs = self.get_next_state(obs, a)
            self.last_obs, self.last_action = obs, a

                return a

        a = self._greedy_action(obs)
        if self.get_next_state(obs, a) == obs:
            # fallback to any unblocked move
            for b in self.get_actions():
                if self.get_next_state(obs, b) != obs:
                    a = b
                    break
        self.last_obs, self.last_action = obs, a
        return a

class GeneratedPolicy:
    """
    Two-stage planner:
    - Stage 1: Plan to the entrance (-1, 0) from wherever we start (in the void).
    - Stage 2: Once inside the maze (row>=0,col>=0), plan to goal.
    Replans only if the cached plan becomes invalid.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.goal = None
        self.stage = 1
        self.plan = []
        self.expected_obs = None

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.stage = 1
        self.plan = []
        self.expected_obs = None

    def _make_successors(self):
        def get_successors(state):
            for a in self.get_actions():
                ns = self.get_next_state(state, a)
                if ns != state:
                    yield (a, ns, float(self.get_cost()))
        return get_successors

    def _heuristic_to(self, target):
        def h(s):
            return float(abs(s[0] - target[0]) + abs(s[1] - target[1]))
        return h

    def _plan(self, start, goal_state, timeout=3.0, max_exp=1_000_000):
        _, actions, _ = run_astar(
            initial_state=start,
            check_goal=lambda s: s == goal_state,
            get_successors=self._make_successors(),
            heuristic=self._heuristic_to(goal_state),
            max_expansions=max_exp,
            max_evals=max_exp,
            timeout=timeout,
            lazy_expansion=True,
        )
        return list(actions)

    def _plan_to_goal(self, start, timeout=4.0, max_exp=2_000_000):
        if self.goal is None:
            return []
        _, actions, _ = run_astar(
            initial_state=start,
            check_goal=lambda s: self.check_goal(s, self.goal),
            get_successors=self._make_successors(),
            heuristic=self._heuristic_to(self.goal),
            max_expansions=max_exp,
            max_evals=max_exp,
            timeout=timeout,
            lazy_expansion=False,
        )
        return list(actions)

    def get_action(self, obs):
        # Follow cached plan if still aligned
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        inside = (r >= 0 and c >= 0)

                self.stage = 2
                self.plan = []
                self.expected_obs = None
            else:
                # Plan to entrance cell (-1,0) (not (0,0)), then a single South will enter maze.
                self.plan = self._plan(obs, (-1, 0), timeout=2.0, max_exp=500_000)

                if self.plan:
                    a = self.plan.pop(0)
                    self.expected_obs = self.get_next_state(obs, a)
                    return a
                # If can't plan (shouldn't happen), fall back: move toward col 0 and row -2 then south.
                if c < 0:
                    return 2
                if c > 0:
                    return 3
                return 1 if r < -1 else 1

        # Stage 2: inside maze, plan to goal.
        if inside:
            self.plan = self._plan_to_goal(obs, timeout=4.0, max_exp=2_000_000)
            self.expected_obs = obs
            if self.plan:
                a = self.plan.pop(0)
                self.expected_obs = self.get_next_state(obs, a)
                return a

        # Fallback: return any unblocked move else 0
        for a in self.get_actions():
            if self.get_next_state(obs, a) != obs:
                return a
        return 0

class GeneratedPolicy:
    """
    Hybrid with "micro-planning":
    - Uses greedy steps most of the time.
    - If greedy is blocked OR doesn't improve Manhattan distance for several steps, run a bounded A*
      to find a short improving path (not necessarily all the way to goal), then follow it.
    This reduces heavy planning while escaping local traps.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal = check_goal

        self.goal = None
        self.plan = []
        self.expected_obs = None

        self.no_improve_steps = 0
        self.last_dist = None

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None
        self.no_improve_steps = 0
        self.last_dist = None if self.goal is None else (abs(obs[0] - self.goal[0]) + abs(obs[1] - self.goal[1]))

    def _dist(self, s):
        if self.goal is None:
            return 0
        return abs(s[0] - self.goal[0]) + abs(s[1] - self.goal[1])

    def _make_successors(self):
        def get_successors(state):
            for a in self.get_actions():
                ns = self.get_next_state(state, a)
                if ns != state:
                    yield (a, ns, float(self.get_cost()))
        return get_successors

    def _greedy(self, obs):
        best = None
        for a in self.get_actions():
            ns = self.get_next_state(obs, a)
            if ns == obs:
                continue
            d = self._dist(ns)
            if best is None or d < best[0]:
                best = (d, a)
        if best is not None:
            return best[1]
        return 0

    def _micro_plan_to_better(self, obs, dist_threshold_improve=2):
        """
        Plan to any state whose distance is at least `dist_threshold_improve` better than current.
        Uses A* with the goal predicate defined as "distance <= current_dist - improve".
        """
        if self.goal is None:
            return []

        current_d = self._dist(obs)
        target_d = max(0, current_d - dist_threshold_improve)

        def is_good(s):
            return self._dist(s) <= target_d

        def heuristic(s):
            # Encourage reaching any "good" state; lower is better.
            # If already good, 0.
            d = self._dist(s)
            return float(max(0, d - target_d))

        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=is_good,
            get_successors=self._make_successors(),
            heuristic=heuristic,
            max_expansions=120_000,
            max_evals=120_000,
            timeout=0.6,
            lazy_expansion=True,
        )
        return list(actions)

    def _outer_void_action(self, obs):
        r, c = obs
        if r == -1 and c != 0:
            return 0 if self.get_next_state(obs, 0) != obs else 1
        if r > -2:
            return 0
        if c < 0:
            return 2
        if c > 0:
            return 3
        return 1

    def get_action(self, obs):
        # Follow cached plan if aligned
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        r, c = obs
        if r < 0 or c < 0:
            return self._outer_void_action(obs)

        # Update "no improvement" counter
        d = self._dist(obs)
        if self.last_dist is None:
            self.last_dist = d
        if d < self.last_dist:
            self.no_improve_steps = 0
        else:
            self.no_improve_steps += 1
        self.last_dist = d

        # Greedy step
        a = self._greedy(obs)
        greedy_next = self.get_next_state(obs, a)

        blocked = (greedy_next == obs)
        if blocked or self.no_improve_steps >= 6:
            # Micro-plan to a state with improved distance
            self.plan = self._micro_plan_to_better(obs, dist_threshold_improve=2)
            self.expected_obs = obs
            if self.plan:
                a = self.plan.pop(0)
                self.expected_obs = self.get_next_state(obs, a)
                return a
            # If micro-plan fails, fall back to full plan to goal (bounded)
            if self.goal is not None:
                _, actions, _ = run_astar(
                    initial_state=obs,
                    check_goal=lambda s: self.check_goal(s, self.goal),
                    get_successors=self._make_successors(),
                    heuristic=lambda s: float(self._dist(s)),
                    max_expansions=600_000,
                    max_evals=600_000,
                    timeout=2.0,
                    lazy_expansion=False,
                )
                self.plan = list(actions)
                self.expected_obs = obs
                if self.plan:
                    a = self.plan.pop(0)
                    self.expected_obs = self.get_next_state(obs, a)
                    return a

        # Ensure non-blocked if possible
        if self.get_next_state(obs, a) == obs:
            for b in self.get_actions():
                if self.get_next_state(obs, b) != obs:
                    return b
        return a if a in [0, 1, 2, 3] else 0

        self.plan = []
        self.expected_obs = None

        self.suffix_cache = {}  # state -> list of actions to goal

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None
        self.suffix_cache = {}

    def _make_successors(self):
        def get_successors(state):
            for a in self.get_actions():
                ns = self.get_next_state(state, a)
                if ns != state:
                    yield (a, ns, float(self.get_cost()))
        return get_successors

    def _h(self, s):
        if self.goal is None:
            return 0.0
        return float(abs(s[0] - self.goal[0]) + abs(s[1] - self.goal[1]))

    def _outer_void_action(self, obs):
        r, c = obs
        if r == -1 and c != 0:
            return 0 if self.get_next_state(obs, 0) != obs else 1
        if r > -2:
            return 0
        if c < 0:
            return 2
        if c > 0:
            return 3
        return 1

    def _cache_suffixes(self, state_seq, action_seq):
        # For each prefix state, store the remaining action suffix.
        # This is safe because environment is static within episode.
        for i, s in enumerate(state_seq):
            if i < len(action_seq):
                self.suffix_cache[s] = list(action_seq[i:])

    def _plan_from(self, obs):
        if self.goal is None:
            return []
        # Check memoized suffix first
        if obs in self.suffix_cache and self.suffix_cache[obs]:
            return list(self.suffix_cache[obs])

        state_seq, action_seq, _ = run_astar(
            initial_state=obs,
            check_goal=lambda s: self.check_goal(s, self.goal),
            get_successors=self._make_successors(),
            heuristic=self._h,
            max_expansions=1_200_000,
            max_evals=1_200_000,
            timeout=3.0,
            lazy_expansion=True,
        )
        actions = list(action_seq)
        if state_seq and actions:
            self._cache_suffixes(state_seq, actions)
        return actions

    def get_action(self, obs):
        # Follow cached plan if aligned
        if self.plan and self.expected_obs == obs:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        r, c = obs
        if r < 0 or c < 0:
            return self._outer_void_action(obs)

        # Plan (or reuse memoized suffix) from current state
        self.plan = self._plan_from(obs)
        self.expected_obs = obs
        if self.plan:
            a = self.plan.pop(0)
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Fallback: any unblocked move
        for a in self.get_actions():
            if self.get_next_state(obs, a) != obs:
                return a
        return 0
