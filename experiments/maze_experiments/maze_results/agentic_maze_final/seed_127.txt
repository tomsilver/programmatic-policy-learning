from prpl_utils.search import run_astar, SearchMetrics


class GeneratedPolicy:
    """
    Purely reactive wall-following / bug-like navigation.

    Strategy:
    - Two phases:
      (1) Outside/entrance acquisition: go to (-2, 0) then to (-1, 0) then into (0, 0).
      (2) Inside maze: left-hand wall follower (with a small bias toward moving
          toward the goal when multiple options exist).
    - No A* usage.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None
        self.mode = "to_entrance"
        self.last_action = 1  # arbitrary init; used for wall-follow order

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.mode = "to_entrance"
        self.last_action = 1

    @staticmethod
    def _manhattan(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _valid(self, s, a):
        ns = self.get_next_state(s, a)
        return ns != s

    def _step_toward(self, obs, target):
        r, c = obs
        tr, tc = target
        # Prefer vertical then horizontal.
        prefs = []
        if tr < r:
            prefs.append(0)
        elif tr > r:
            prefs.append(1)
        if tc > c:
            prefs.append(2)
        elif tc < c:
            prefs.append(3)
        # Fill with remaining actions as fallback.
        for a in [0, 1, 2, 3]:
            if a not in prefs:
                prefs.append(a)
        for a in prefs:
            if self._valid(obs, a):
                return a
        return 0  # must return something valid; if all blocked, 0 returns self-state.

    def _outside_to_entrance(self, obs):
        # Target staging point is (-2, 0) to safely approach entrance from above.
        if obs == (-2, 0):
            # Move into entrance cell (-1, 0) if possible, else wait (should be possible).
            return 1 if self._valid(obs, 1) else self._step_toward(obs, (-3, 0))
        if obs == (-1, 0):
            # Step into maze (0,0)
            return 1 if self._valid(obs, 1) else 0
        # Move toward (-2,0).
        return self._step_toward(obs, (-2, 0))

    def _inside_wall_follow(self, obs):
        # Left-hand rule relative to heading = last_action.
        # Map: 0N 1S 2E 3W. Left turn mapping:
        left_of = {0: 3, 3: 1, 1: 2, 2: 0}
        right_of = {0: 2, 2: 1, 1: 3, 3: 0}
        back_of = {0: 1, 1: 0, 2: 3, 3: 2}

        forward = self.last_action
        left = left_of[forward]
        right = right_of[forward]
        back = back_of[forward]

        candidates = [left, forward, right, back]

        # If goal known, slightly bias toward actions that reduce Manhattan distance.
        if self.goal is not None:
            scored = []
            for a in candidates:
                ns = self.get_next_state(obs, a)
                if ns != obs:
                    scored.append((self._manhattan(ns, self.goal), a))
                else:
                    scored.append((10**9, a))
            scored.sort()
            # Keep left-hand preference, but take the best among the first two if close.
            best_a = scored[0][1]
            if self._valid(obs, best_a):
                return best_a

        for a in candidates:
            if self._valid(obs, a):
                return a
        return 0

    def get_action(self, obs):
        # Mode switch once we have entered the maze.
        if obs[0] >= 0 and obs[1] >= 0:
            self.mode = "in_maze"

        if self.mode == "to_entrance":
            a = self._outside_to_entrance(obs)
        else:
            a = self._inside_wall_follow(obs)

        # Update last_action only if action actually moves; otherwise keep heading.
        if self.get_next_state(obs, a) != obs:
            self.last_action = a
        return int(a)

    """

        self.get_next_state = get_next_state

    def _manhattan(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _make_successors(self):
        def get_successors(state):
            for action in self.get_actions():
                ns = self.get_next_state(state, action)
                if ns != state:
                    yield (action, ns, float(self.get_cost()))
        return get_successors

    def _heuristic(self, state):
        if self.goal is None:
            return 0.0
        return float(self._manhattan(state, self.goal))

            self.plan = []
            self.expected_obs = obs
            return

            initial_state=obs,
            check_goal=lambda s: self.check_goal_env(s, self.goal),
            get_successors=self._make_successors(),
            heuristic=self._heuristic,
            max_expansions=2_000_000,
            max_evals=4_000_000,
            timeout=2.0,
            lazy_expansion=False,
        )
        self.metrics = metrics
        self.plan = list(actions) if actions is not None else []
        self.expected_obs = obs

    def get_action(self, obs):
        # Follow cached plan only if perfectly synchronized.
        if self.plan and self.expected_obs == obs:
            a = int(self.plan.pop(0))
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Otherwise replan from current obs.
        self._plan_from(obs)

        # If planning yielded a plan, take its first step.
        if self.plan and self.expected_obs == obs:
            a = int(self.plan.pop(0))
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Fallback: return any valid action (or 0 if stuck).
        for a in self.get_actions():
            if self.get_next_state(obs, a) != obs:
                return int(a)
        return 0

class GeneratedPolicy:
    """
    Hybrid with "stuck detection":
    - Default: greedy Manhattan descent toward goal (works well in open corridors).
    - If progress stalls (no reduction in heuristic for K steps) or a move is blocked
      repeatedly, invoke A* to escape and cache the plan.
    - Outside: deterministic approach to entrance first.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None
        self.plan = []
        self.expected_obs = None

        self.inside = False
        self.no_improve = 0
        self.last_h = None
        self.blocked_moves = 0

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None

        self.inside = (obs[0] >= 0 and obs[1] >= 0)
        self.no_improve = 0
        self.last_h = None
        self.blocked_moves = 0

    @staticmethod
    def _manhattan(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _h(self, s):
        if self.goal is None:
            return 0
        return self._manhattan(s, self.goal)

    def _valid(self, s, a):
        return self.get_next_state(s, a) != s

    def _outside_step(self, obs):
        # Go to (-2,0) then into (-1,0) then into (0,0)
        if obs == (-1, 0):
            return 1 if self._valid(obs, 1) else 0
        if obs == (-2, 0):
            return 1 if self._valid(obs, 1) else 0
        # Greedy toward (-2,0)
        target = (-2, 0)
        r, c = obs
        tr, tc = target
        prefs = []
        if tr < r:
            prefs.append(0)
        elif tr > r:
            prefs.append(1)
        if tc > c:
            prefs.append(2)
        elif tc < c:
            prefs.append(3)
        for a in [0, 1, 2, 3]:
            if a not in prefs:
                prefs.append(a)
        for a in prefs:
            if self._valid(obs, a):
                return a
        return 0

    def _make_successors(self):
        def get_successors(state):
            for action in self.get_actions():
                ns = self.get_next_state(state, action)
                if ns != state:
                    yield (action, ns, float(self.get_cost()))
        return get_successors

        return float(self._h(s))

    def _plan_from(self, obs, timeout=1.0, max_exp=800_000):
        if self.goal is None:
            self.plan = []
            self.expected_obs = obs
            return
        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=lambda s: self.check_goal_env(s, self.goal),
            get_successors=self._make_successors(),
            heuristic=self._heuristic,
            max_expansions=max_exp,
            max_evals=2 * max_exp,
            timeout=timeout,
            lazy_expansion=True,
        )
        self.plan = list(actions) if actions is not None else []
        self.expected_obs = obs

    def _greedy_step(self, obs):
        if self.goal is None:
            for a in self.get_actions():
                if self._valid(obs, a):
                    return int(a)
            return 0

        # Choose action that minimizes heuristic; tie-break by preferring S/E.
        best = None
        bestv = 10**18
        for a in [1, 2, 0, 3]:
            ns = self.get_next_state(obs, a)
            if ns == obs:
                continue
            v = self._h(ns)
            if v < bestv:
                bestv = v
                best = a
        if best is not None:
            return int(best)

        # If all blocked, return 0.
        return 0

        if not self.inside:
            return int(self._outside_step(obs))

        if self.plan and self.expected_obs == obs:
            a = int(self.plan.pop(0))
            self.expected_obs = self.get_next_state(obs, a)
            return a

            # Any mismatch invalidates the plan.
            self.plan = []
            self.expected_obs = None

        # Reactive step
        a = int(self._greedy_step(obs))
        ns = self.get_next_state(obs, a)
        if ns == obs:
            self.blocked_moves += 1
        else:
            self.blocked_moves = 0

        # Progress tracking
        hnow = self._h(obs)
        hnext = self._h(ns) if ns != obs else hnow
        if self.last_h is None:
            self.last_h = hnow
        if hnext < hnow:
            self.no_improve = 0
        else:
            self.no_improve += 1
        self.last_h = hnext

        # Trigger planning if stuck.
        if self.blocked_moves >= 2 or self.no_improve >= 8:
            self._plan_from(obs, timeout=1.2, max_exp=1_200_000)
            self.no_improve = 0
            self.blocked_moves = 0
            if self.plan and self.expected_obs == obs:
                a2 = int(self.plan.pop(0))
                self.expected_obs = self.get_next_state(obs, a2)
                return a2

        # Otherwise take reactive action (even if it doesn't improve).
        if ns != obs:
            return a

        # Final fallback: any valid action
        for aa in self.get_actions():
            if self._valid(obs, aa):
                return int(aa)
        return 0

class GeneratedPolicy:
    """
    Two-stage planning:
    - Stage A: Ensure we enter the maze by explicitly planning to the entrance cell (-1,0)
      if not already inside, using A* (robust from arbitrary void starts).
    - Stage B: Once inside, plan to goal with A* and cache full plan.

    This is more planning-heavy than needed outside, but very robust to odd starts.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None
        self.plan = []
        self.expected_obs = None
        self.stage = "to_entrance"  # or "to_goal"

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.plan = []
        self.expected_obs = None
        self.stage = "to_goal" if (obs[0] >= 0 and obs[1] >= 0) else "to_entrance"

    def _make_successors(self):
        def get_successors(state):
            for action in self.get_actions():
                ns = self.get_next_state(state, action)
                if ns != state:
                    yield (action, ns, float(self.get_cost()))
        return get_successors

    @staticmethod
    def _manhattan(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _heuristic_to(self, target):
        def h(s):
            return float(self._manhattan(s, target))
        return h

    def _plan_to(self, obs, target, timeout, max_exp):
        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=lambda s: s == target,
            get_successors=self._make_successors(),
            heuristic=self._heuristic_to(target),
            max_expansions=max_exp,
            max_evals=2 * max_exp,
            timeout=timeout,
            lazy_expansion=False,
        )
        self.plan = list(actions) if actions is not None else []
        self.expected_obs = obs

    def _plan_to_goal(self, obs, timeout, max_exp):
        if self.goal is None:
            self.plan = []
            self.expected_obs = obs
            return
        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=lambda s: self.check_goal_env(s, self.goal),
            get_successors=self._make_successors(),
            heuristic=lambda s: float(self._manhattan(s, self.goal)),
            max_expansions=max_exp,
            max_evals=2 * max_exp,
            timeout=timeout,
            lazy_expansion=True,
        )
        self.plan = list(actions) if actions is not None else []
        self.expected_obs = obs

    def get_action(self, obs):
        # Stage transitions
        if obs[0] >= 0 and obs[1] >= 0:
            self.stage = "to_goal"

        # Follow cached plan if synced.
        if self.plan and self.expected_obs == obs:
            a = int(self.plan.pop(0))
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # Otherwise (re)plan based on stage.
        if self.stage == "to_entrance":
            # Plan to entrance (-1,0), then next will step South into (0,0) via replans.
            self._plan_to(obs, (-1, 0), timeout=0.6, max_exp=250_000)
        else:
            self._plan_to_goal(obs, timeout=1.2, max_exp=1_200_000)

        if self.plan and self.expected_obs == obs:
            a = int(self.plan.pop(0))
            self.expected_obs = self.get_next_state(obs, a)
            return a

        # If A* failed, use a safe deterministic outside approach or a generic fallback.
        if self.stage == "to_entrance":
            # Move toward (-2,0) then to (-1,0)
            if obs == (-1, 0):
                return 1
            if obs[1] != 0:
                a = 3 if obs[1] > 0 else 2
                return int(a)
            if obs[0] != -2:

from prpl_utils.search import run_astar, SearchMetrics


class GeneratedPolicy:
    """
    Hybrid "corridor follower" + A*:
    - Outside: deterministic entrance acquisition.
    - Inside: if the local structure is corridor-like (<=1 forward choice excluding back),
      follow it reactively with a goal-directed tie-break.
    - If at an intersection, dead-end, or after oscillation is detected, run A* and cache.

    This reduces planning calls in long corridors but uses A* when decisions matter.
    """

    def __init__(self, get_actions, get_next_state, get_cost, check_goal):
        self.get_actions = get_actions
        self.get_next_state = get_next_state
        self.get_cost = get_cost
        self.check_goal_env = check_goal

        self.goal = None
        self.inside = False

        self.plan = []
        self.expected_obs = None

        self.prev_obs = None
        self.prev_prev_obs = None
        self.last_action = 1

    def reset(self, obs, info):
        self.goal = info.get("goal", None)
        self.inside = (obs[0] >= 0 and obs[1] >= 0)

        self.plan = []
        self.expected_obs = None

        self.prev_obs = None
        self.prev_prev_obs = None
        self.last_action = 1

    def _valid(self, s, a):
        return self.get_next_state(s, a) != s

    @staticmethod
    def _manhattan(a, b):
        return abs(a[0] - b[0]) + abs(a[1] - b[1])

    def _outside_step(self, obs):
        if obs == (-1, 0):
            return 1 if self._valid(obs, 1) else 0
        if obs == (-2, 0):
            return 1 if self._valid(obs, 1) else 0
        # Move to (-2,0): adjust col first, then row.
        if obs[1] != 0:
            a = 3 if obs[1] > 0 else 2
            if self._valid(obs, a):
                return a
        if obs[0] != -2:
            a = 0 if obs[0] > -2 else 1
            if self._valid(obs, a):
                return a
        for a in [0, 1, 2, 3]:
            if self._valid(obs, a):
                return a
        return 0

    def _make_successors(self):
        def get_successors(state):
            for action in self.get_actions():
                ns = self.get_next_state(state, action)
                if ns != state:
                    yield (action, ns, float(self.get_cost()))
        return get_successors

    def _heuristic(self, s):
        if self.goal is None:
            return 0.0
        return float(self._manhattan(s, self.goal))

    def _plan_from(self, obs):
        if self.goal is None:
            self.plan = []
            self.expected_obs = obs
            return
        _, actions, _ = run_astar(
            initial_state=obs,
            check_goal=lambda s: self.check_goal_env(s, self.goal),
            get_successors=self._make_successors(),
            heuristic=self._heuristic,
            max_expansions=900_000,
            max_evals=1_800_000,
            timeout=1.0,
            lazy_expansion=True,
        )
        self.plan = list(actions) if actions is not None else []
        self.expected_obs = obs

    def _neighbors(self, obs):
        moves = []
        for a in [0, 1, 2, 3]:
            ns = self.get_next_state(obs, a)
            if ns != obs:
                moves.append((a, ns))
        return moves

    def _corridor_reactive(self, obs):
        moves = self._neighbors(obs)
        if not moves:
            return 0

        back_of = {0: 1, 1: 0, 2: 3, 3: 2}
        back = back_of[self.last_action]

        forwardish = [(a, ns) for (a, ns) in moves if a != back]
        # Corridor-like if only 1 "forward" option (or none, i.e., forced to backtrack).
        if len(forwardish) <= 1:
            cand = forwardish if forwardish else moves
            if self.goal is None:
                return int(cand[0][0])
            # choose among candidates by heuristic
            best = min(cand, key=lambda x: self._manhattan(x[1], self.goal))
            return int(best[0])

        # Not a corridor: signal to plan.
        return None

    def get_action(self, obs):
        self.inside = self.inside or (obs[0] >= 0 and obs[1] >= 0)
        if not self.inside:
            a = int(self._outside_step(obs))
            if self.get_next_state(obs, a) != obs:
                self.last_action = a
            return a

        # Oscillation detection: obs == prev_prev_obs indicates back-and-forth.
        oscillating = (self.prev_prev_obs is not None and obs == self.prev_prev_obs)

        # Follow cached plan if synced and not oscillating.
        if not oscillating and self.plan and self.expected_obs == obs:
            a = int(self.plan.pop(0))
            self.expected_obs = self.get_next_state(obs, a)
            if self.get_next_state(obs, a) != obs:
                self.last_action = a
            self.prev_prev_obs, self.prev_obs = self.prev_obs, obs
            return a

        # Invalidate plan on mismatch/oscillation.
        self.plan = []
        self.expected_obs = None

        # Try corridor reactive; if returns None, plan.
        a = self._corridor_reactive(obs)
        if a is None or oscillating:
            self._plan_from(obs)
            if self.plan and self.expected_obs == obs:
                a = int(self.plan.pop(0))
                self.expected_obs = self.get_next_state(obs, a)
            else:
                # Fallback: pick any valid move, prefer toward goal.
                moves = self._neighbors(obs)
                if moves:
                    if self.goal is not None:
                        a = int(min(moves, key=lambda x: self._manhattan(x[1], self.goal))[0])
                    else:
                        a = int(moves[0][0])
                else:
                    a = 0

        if self.get_next_state(obs, a) != obs:
            self.last_action = int(a)
        self.prev_prev_obs, self.prev_obs = self.prev_obs, obs
        return int(a)
